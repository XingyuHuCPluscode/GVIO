function(apply_msvc_cuda_flags target)
  if(MSVC)
    target_compile_options(${target} PRIVATE
      $<$<COMPILE_LANGUAGE:CXX>:/bigobj /EHsc /Zi /Ob0 /utf-8>
    )
  endif()
endfunction()
